const { spawn } = require("child_process");

const APP_DIR = __dirname;
const SERVER_ENTRY = `${APP_DIR}/server.js`;
const APP_PING = "http://127.0.0.1:8000/api/ping";
const OLLAMA_TAGS = "http://127.0.0.1:11434/api/tags";

let startingServer = false;
let startingOllama = false;

async function isAlive(url) {
  try {
    const response = await fetch(url, { method: "GET" });
    return response.ok;
  } catch (error) {
    return false;
  }
}

function spawnDetached(command, args, logName) {
  const child = spawn(command, args, {
    cwd: APP_DIR,
    detached: true,
    stdio: "ignore"
  });
  child.unref();
  console.log(`[watchdog] spawned ${logName}`);
}

async function ensureServer() {
  const alive = await isAlive(APP_PING);
  if (alive || startingServer) return;
  startingServer = true;
  try {
    spawnDetached("node", [SERVER_ENTRY], "server.js");
  } finally {
    setTimeout(() => {
      startingServer = false;
    }, 4000);
  }
}

async function ensureOllama() {
  const alive = await isAlive(OLLAMA_TAGS);
  if (alive || startingOllama) return;
  startingOllama = true;
  try {
    spawnDetached("ollama", ["serve"], "ollama serve");
  } finally {
    setTimeout(() => {
      startingOllama = false;
    }, 4000);
  }
}

async function tick() {
  await Promise.all([ensureServer(), ensureOllama()]);
}

console.log("[watchdog] started");
tick();
setInterval(tick, 5000);

Что это

Простой GPT-подобный чат-бот, написанный с нуля на Python без ML-библиотек. Работает на основе словаря слов и алгоритма схожести по буквам.

Архитектура

ФайлРоль
GPT.pyUI-оболочка (Pythonista ui + WKWebView)
INIT.pyЯдро: learn(text) и answer(text)
Functions.pyУтилиты: очистка текста, работа со словарями, веса
Root_model.pyАлгоритм схожести двух слов по позициям букв
Base_weight.pyРанжирование слов по частоте (bubble-sort-like)
Dictionary.pyХранение словаря слов (Dictionary.pkl)

Как работает

  1. learn(text) — добавляет новые слова из текста в словарь
  2. answer(text) — для каждого слова вопроса считает вес и схожесть со всеми словами словаря, возвращает самое близкое слово
  3. Вес слова растёт каждый раз, когда оно используется в запросе

import pickle
import os

class WordWeight:
	def __init__(self, text, filename="GPT.pkl"):
		self.filename = filename
		self.text = self.clean(text)
		self.weight = self.load()
		
	# очистка текста
	def clean(self, text):
		for z in [":", ",", ".", "!", "?", ";", "«", "»", ")", "(", "-"]:
			text = text.replace(z, "")
		return text.lower().split(" ")
		
	# загрузка списка рангов
	def load(self):
		if os.path.exists(self.filename):
			try:
				with open(self.filename, "rb") as f:
					return pickle.load(f)
			except:
				pass
				
		with open(self.filename, "wb") as f:
			pickle.dump([], f)
		return []
		
	# сохранение
	def save(self):
		with open(self.filename, "wb") as f:
			pickle.dump(self.weight, f)

	# поднять слово на одну позицию вверх
	def promote_one(self, word):
		# если нет — добавить в конец
		if word not in self.weight:
			self.weight.append(word)
			self.save()
			return
			
		idx = self.weight.index(word)
		
		# если слово уже на первом месте — ничего не делать
		if idx == 0:
			return
			
		# меняем местами слово и элемент выше
		self.weight[idx], self.weight[idx - 1] = self.weight[idx - 1], self.weight[idx]
		
		self.save()
		
	# обновить весь текст
	def update(self):
		for w in self.text:
			self.promote_one(w)
			
	# получить позицию слова
	def get_position(self, word):
		if word in self.weight:
			return self.weight.index(word)
		else:
			return None
		
	# основной метод
	def run(self):
		word = self.text[0]
		self.update()
		return self.get_position(word)+1

Что это

Настольная словесная игра с элементами бродилки для 2–4 игроков на одном устройстве. Игроки составляют слова из букв выпавшей карточки, зарабатывают очки и двигают фишки по замкнутому маршруту из 48 клеток. Первый до 50-й клетки — побеждает.

Как работает

Экраны
  1. Экран загрузки — логотип, фоновая загрузка словаря
  2. Диалог — выбор количества игроков (2/3/4)
  3. Главное поле — маршрут с фишками, текущий ход, очки
  4. Экран-заглушка — "Синий игрок, нажмите чтобы начать" (защита от подглядывания)
  5. Экран задания — буквы карточки, поле ввода слова, таймер 60 сек
  6. Экран победы — затемнение поля + "[ЦВЕТ] ВЫИГРАЛ"
Структура хода
  1. Тянется карточка из колоды
  2. Если спецкарточка — сохраняется у игрока, тянется следующая
  3. Если у игрока есть спецкарточки — выбор: использовать или играть обычно
  4. Гонщик → +N очков, ход заканчивается
  5. Судья → выбрать жертву, +N себе, -N жертве
  6. Обычная игра → экран задания, 60 сек, составить N слов
  7. Успех → +N очков, фишка вперёд; провал → -N очков, фишка назад
  8. Ход переходит к следующему игроку

N = количество слов (сейчас = 2, в будущем — бросок кубика 1–6)

Игровое поле

  • 48 клеток, замкнутый маршрут, движение по часовой стрелке
  • Старт — правый верх; вниз → влево → вверх → к старту
  • Фишки анимировано двигаются по одной клетке, задержка 0.4 сек
  • Две фишки на одной клетке — одна смещается вниз, не "бьют" друг друга
  • Поле: assets/field.png, маршрут — 48 координат вручную

Экран задания

Что видит игрок:

  • Буквы исходного слова — каждая отдельная кнопка
  • Поле набираемого слова — над кнопками
  • Кнопка OK, кнопка ❌ (сброс)
  • Счётчик слов: 1/2
  • Таймер: обратный отсчёт 60⏰
  • Список использованных слов (справа)

Правила:

  • Буквы нажимаются по одной, нажатая исчезает с поля
  • Минимум 2 буквы
  • Слово должно быть в словаре
  • Нельзя использовать исходное слово
  • Нельзя повторять уже составленные в этом ходу слова
  • ❌ возвращает все буквы обратно

Система очков

  • Очки = номер клетки фишки (score = position)
  • Успех: +N, фишка вперёд
  • Провал: -N, фишка назад (ниже 0 — остаётся на месте)
  • Победа: достичь клетки 50

Словарь

  • Файл: dictionary.txt
  • Слова построчно, длина 7–21 символ
  • Проверка: точное совпадение со строкой из словаря

import SwiftUI

class WebViewCoordinator: NSObject, WKNavigationDelegate {
    let loader: AppLoader

    init(loader: AppLoader) {
        self.loader = loader
    }

    func webView(_ webView: WKWebView, didFinish navigation: WKNavigation!) {
        Task {
            await injectDictionary(into: webView)
        }
    }

    private func injectDictionary(into webView: WKWebView) async {
        guard let dictText = await loader.getDictionaryText() else { return }
        // Экранируем для безопасной передачи в JS-строку
        let escaped = dictText
            .replacingOccurrences(of: "\\", with: "\\\\")
            .replacingOccurrences(of: "`", with: "\\`")
        let js = "window.__injectDictionary(`\(escaped)`);"
        await MainActor.run {
            webView.evaluateJavaScript(js, completionHandler: nil)
        }
    }
}

struct ContentView: UIViewRepresentable {
    let loader: AppLoader

    func makeCoordinator() -> WebViewCoordinator {
        WebViewCoordinator(loader: loader)
    }

    func makeUIView(context: Context) -> WKWebView {
        let configuration = WKWebViewConfiguration()
        configuration.allowsInlineMediaPlayback = true
        configuration.mediaTypesRequiringUserActionForPlayback = []

        let webView = WKWebView(frame: .zero, configuration: configuration)
        webView.scrollView.isScrollEnabled = true
        webView.scrollView.bounces = false
        webView.navigationDelegate = context.coordinator

        Task {
            await loadApp(into: webView)
        }

        return webView
    }

    func updateUIView(_ uiView: WKWebView, context: Context) {}

    private func loadApp(into webView: WKWebView) async {
        guard let indexURL = await loader.getIndexURL() else {
            await MainActor.run {
                _ = webView.loadHTMLString("<h1>❌ index.html не найден</h1>", baseURL: nil)
            }
            return
        }

        guard let appFolder = await loader.getAppFolderURL() else {
            await MainActor.run {
                _ = webView.loadHTMLString("<h1>❌ Папка приложения не найдена</h1>", baseURL: nil)
            }
            return
        }

        await MainActor.run {
            _ = webView.loadFileURL(indexURL, allowingReadAccessTo: appFolder)
        }
    }
}

// MARK: - AppLoader
// Загружает HTML-приложение из папки VocabularyRace

@MainActor
class AppLoader {

    private let gameFolder = "GameWeb"
    private let documentsURL = FileManager.default.urls(for: .documentDirectory, in: .userDomainMask).first!

    func getIndexURL() async -> URL? {
        let documentsGameFolder = documentsURL.appendingPathComponent(gameFolder)
        let documentsIndex = documentsGameFolder.appendingPathComponent("index.html")

        if FileManager.default.fileExists(atPath: documentsIndex.path) {
            return documentsIndex
        }

        if let bundleIndex = Bundle.main.url(forResource: "index", withExtension: "html", subdirectory: gameFolder) {
            return bundleIndex
        }

        if let bundleFolder = Bundle.main.url(forResource: gameFolder, withExtension: nil) {
            do {
                try FileManager.default.copyItem(at: bundleFolder, to: documentsGameFolder)
                return documentsIndex
            } catch {
                print("❌ Ошибка копирования: \(error)")
            }
        }

        return nil
    }

    func getAppFolderURL() async -> URL? {
        let documentsGameFolder = documentsURL.appendingPathComponent(gameFolder)
        if FileManager.default.fileExists(atPath: documentsGameFolder.path) {
            return documentsGameFolder
        }
        return Bundle.main.url(forResource: gameFolder, withExtension: nil)
    }

    func getDictionaryText() async -> String? {
        // Сначала ищем в Documents (если уже скопировали)
        let documentsGameFolder = documentsURL.appendingPathComponent(gameFolder)
        let documentsDict = documentsGameFolder.appendingPathComponent("dictionary.txt")
        if let text = try? String(contentsOf: documentsDict, encoding: .utf8) {
            return text
        }
        // Затем в бандле
        if let bundleURL = Bundle.main.url(forResource: "dictionary", withExtension: "txt", subdirectory: gameFolder),
           let text = try? String(contentsOf: bundleURL, encoding: .utf8) {
            return text
        }
        return nil
    }
}

Проблема

Большинство трансформеров имеют фиксированный сценарий трансформаций — ребёнок повторяет, а не придумывает. Это ограничивает воображение.

Идея

Звезда Луки — трансформер без правил. Цельный объект на шаровых шарнирах: одно движение — скрутил, повернул — и форма изменилась. Нет инструкции — есть бесконечные варианты.

Принцип

Базовая форма — плоский крест (+): каркас из 4 чёрных Г-образных коннекторов между внутренним и внешним квадратом, центр пустой. По периметру каркаса расположены 4 бирюзовых блока, на каждом — шарнир и Г-рычаг, лежащий плашмя. Объект меняет форму через два параметра: поворот Г-рычагов и опциональное скручивание секций. X-форма — это свёрнутое состояние, не базовое. Тот же объект в новом положении становится другой фигурой.

Принцип работы

Все детали соединены через шаровые шарниры (ball-and-socket joint), что обеспечивает свободное вращение в любой плоскости. Объект цельный — детали не снимаются и не переставляются. Трансформация происходит через поворот шарниров; скручивание секций каркаса возможно, но не обязательно.

Компоненты

Ядро (Core)
  • Каркас образует два условных квадрата: внутренний (пустой проём) и внешний (по периметру конструкции)
  • 4 чёрных цельных Г-образных коннектора соединяют внутренний и внешний квадрат, огибая центр и не касаясь друг друга
  • 4 бирюзовых (teal) LEGO Technic блока расположены по периметру внешнего квадрата (точки крепления шарниров)
  • Центр конструкции — пустой: пересечений и центрального хаба нет
  • 4 шаровых шарнира с синими штифтами — на внешнем конце каждого бирюзового блока, точки крепления Г-рычагов
Конечности (Limbs)
  • 2 белых изогнутых рычага — верхние/боковые позиции
  • 2 чёрных изогнутых рычага — нижние/боковые позиции
  • Синие стопорные штифты — фиксируют рычаги в шарнирах
Дополнительные элементы (на части фото)
  • Серая/чёрная квадратная пластина — используется как голова, корпус или хвост
  • Белые прямоугольные плиты — крылья или панели корпуса

Базовая (официальная) форма

Плоский крест (+): все 4 Г-рычага симметрично развёрнуты в стороны, объект лежит в одной плоскости. Это стартовая точка — из неё начинается любая трансформация.

Характеристики

  • Материал: ABS пластик (LEGO)
  • Цвета: бирюзовый, чёрный, белый, синий (штифты)
  • Количество деталей: ~20-30 элементов
  • Возраст: 6+
  • Размер базовой формы: ~10x10x12 см

Механизм трансформации

Форма определяется двумя независимыми параметрами:

  1. Положение Г-образных рычагов — поворот каждого рычага в шарнире задаёт направление конечности (вверх, вниз, в сторону, вперёд). Это основной параметр, всегда задействован.
  2. Скручивание секций — поворот частей ядра относительно друг друга. Опционально: многие формы достигаются без скручивания, только поворотом Г-рычагов.

Итого: форму можно получить только поворотом рычагов (без скручивания) или поворотом рычагов + скручиванием вместе. Это даёт два класса трансформаций.

Геометрия каркаса (вид сверху)

  • Внутренний квадрат — пустой проём, через него видно фон
  • Внешний квадрат — условная граница каркаса по бирюзовым блокам
  • 4 чёрных Г-коннектора соединяют внутренний и внешний квадрат по углам, оставляя зазоры между собой

Ограничения и устойчивость

  • Шарниры держат форму за счёт трения; при частых трансформациях нужна проверка износа
  • Скручивание допускается, но не является обязательным — часть форм строится без него

Сборка (черновая последовательность)

  1. Собрать кольцевой каркас: 4 чёрных Г-коннектора между внутренним и внешним квадратом
  2. Закрепить 4 бирюзовых блока по периметру каркаса
  3. Вставить шаровые шарниры с синими штифтами на внешние стороны бирюзовых блоков
  4. Закрепить 4 Г-рычага (2 белых, 2 чёрных)

Состав конструкции: 16 деталей

1. Бирюзовые блоки (× 4)
  • Расположение: центральный кластер 2×2
  • Функция: несущий узел; принимает шар сверху и соединяется с Г-коннектором сбоку
  • Сверху блока — гнездо для шара (отдельная деталь)
  • Цвет: бирюзовый / тёмно-зелёный
2. Шары — дисковидные шарниры (× 4)
  • Форма: дисковидная / полусферическая (не полный шар — чашеобразная форма)
  • Расположение: по одному сверху на каждом бирюзовом блоке
  • Крепление: вставляются штырём в верхнее отверстие бирюзового блока (штырь скрыт внутри, снаружи не виден)
  • Функция: шаровой шарнир — рычаг касается шара своей внешней (вогнутой) стороной, фиксируясь на нём без полного охвата; даёт свободу движения во всех направлениях
  • Цвет: чёрный
3. Чёрные Г-коннекторы (× 4)
  • Форма: дугообразная (плавная кривая, не угловая)
  • Функция: соединяют соседние бирюзовые блоки — каждый коннектор = одна дуга между двумя соседними блоками
  • 4 дуги образуют замкнутое кольцо вокруг центра
  • Внутри кольца — пустой центр (каркасная конструкция)
  • Цвет: чёрный
4. Рычаги (× 4) — 2 белых + 2 чёрных
  • Все 4 одинаковой формы, различаются только цветом
  • Крепление: через шар-шарнир, сбоку бирюзового блока
  • Форма: асимметричный Г (одно плечо длиннее другого)
  • Короткое плечо → идёт наружу, это «рука/нога» объекта (~3.5 см)
  • Длинное плечо → внешней (вогнутой) стороной касается шара-шарнира
  • Угол между плечами Г: приблизительно 90°
  • Расположение в базовой позиции:
  • Белые (2 шт.) — горизонтально, влево и вправо
  • Чёрные (2 шт.) — вертикально, вверх и вниз

Есть моменты когда всё складывается само. Мяч летит точно куда нужно. Нужный человек пишет в нужный момент. Движение выходит идеально — хотя так раньше не получалось. Это называют удачей. Но я думаю это не случайность.

---

Что происходит

Попробуй вспомнить свой лучший момент — в спорте, в разговоре, в игре, в чём угодно. Скорее всего ты в тот момент не думал. Просто делал. Голова была пуста, и всё шло само.

А теперь вспомни когда что-то важное не получалось. Почти наверняка — слишком много думал. Анализировал, сомневался, боялся ошибиться.

Я заметил закономерность: чем меньше мыслей — тем лучше результат. И это работает не иногда. Это работает всегда.

---

Как это устроено

Всё дело в рефлексах.

Подсознание — это не что-то мистическое. Это накопленные паттерны, которые работают быстро и точно без участия сознания. Твоё тело умеет очень многое. Проблема в том что сознание постоянно мешает этому проявиться — анализирует, добавляет сомнения, тормозит.

Когда голова пуста — рефлексы получают управление. И делают всё лучше.

Это три слоя:

Шум — обычное состояние. Мысли перекрывают рефлексам дорогу.

Тишина — состояние Ниндзя. Мысли выгнаны. Подсознание выходит на первый план.

Поток — результат. Всё идёт точно и само. Это и воспринимается как удача.

---

Это не теория — это практика

Вот что реально происходило:

Воздушный змей. Сильный ветер, сложные условия. Я вошёл в состояние — и змей буквально затянуло прямо в руки. Не через усилие, а через отсутствие лишних мыслей.

Футбол. Я плохо обращаюсь с мячом. Но однажды на несколько секунд включилось что-то другое — и игра была на уровне выше моего. Потом прошло. Ровно когда начал думать.

Компьютерная игра. Я намеренно "отдал управление" Ниндзя — перестал думать о том что делать. Персонаж заиграл намного лучше.

Кунг-фу на пляже. В состоянии Ниндзя тело само давало мне упражнения — странные, которые я бы не придумал. После них появлялись новые способности. Подсознание выступало не исполнителем, а автором программы.

---

Три способа использовать

Я нашёл три режима:

Момент — войти в состояние прямо сейчас. Всё идёт лучше пока ты в нём.

Обучение — войти, сделать что-то, выйти. Тело запоминает паттерн через подсознание — быстрее чем через обычные повторения. Навык записывается напрямую в рефлексы.

Генерация — войти и ничего не планировать. Подсознание само придумает что делать. Иногда это открывает то, до чего сознание никогда бы не дошло.

Как войти

Два метода:

Пение в Ниндзя — петь без мыслей. Пение занимает речевой центр мозга, внутренний диалог замолкает. Проще всего начать с этого.

Концентрация на ничто — активно выгонять мысли. Сложнее, работает напрямую. Занимает 1–2 минуты.

Состояние заканчивается само — как только начинаешь думать.

Одно предупреждение

После выхода из состояния примерно в 75% случаев наступает временная полоса невезения. Почему — я пока не знаю. Возможно сознание после отключения начинает мешать сильнее чем обычно. Это проходит.

Это не новая идея — но у неё не было метода

Учёные давно знают что подсознание мощнее сознания при обработке информации. Спортивные тренеры учат "отключать голову" — и результат сразу растёт. В даосизме это называется wu wei — действие через неделание.

Но нигде не было конкретного метода как туда попасть. И нигде не был описан режим обучения — когда подсознание не просто действует, а записывает новые паттерны в тело.

Механика Ниндзя

Механика Ниндзя — это состояние пустой головы, в котором подсознание берёт управление и делает всё лучше чем сознание.

Удача приходит не случайно — она приходит когда перестаёшь мешать себе думать.

---

Проблема

Почему одни и те же действия иногда работают блестяще, а иногда нет? Почему лучшее происходит не когда стараешься, а когда расслаблен? Что за состояние это вызывает и можно ли входить в него намеренно?

---

Как устроено

Есть три слоя:

Шум — обычное состояние. Сознание перегружено мыслями: анализ, сомнения, страх ошибки. Рефлексы заблокированы — они есть, но не могут проявиться.

Тишина — состояние Ниндзя. Мысли выгнаны. Голова пуста. Сознание отходит в сторону и открывает прямой доступ к подсознанию. Рефлексы получают управление.

Поток — результат. Рефлексы работают быстро и точно, без фильтра страха и анализа. Это и воспринимается как удача или сверхрезультат.

---

Почему это работает

Подсознание — это рефлексы: накопленные паттерны которые работают быстро, точно и без участия сознания. Они уже знают что делать. Проблема в том что сознание постоянно перекрывает им дорогу — анализирует, сомневается, мешает.

Когда голова пуста — рефлексы выходят на первый план и делают всё сами.

Это подтверждается наукой: Unconscious Thought Theory (Дейкстерхёйс) показала что подсознание обрабатывает больше переменных одновременно чем сознание. В спорте это называется "зона" — тренеры специально учат отключать голову, потому что результат сразу вырастает.

---

Как войти

Два метода:

  • Пение в Ниндзя — петь без мыслей. Пение занимает речевой центр мозга, внутренний диалог замолкает. Легче чем второй метод.
  • Концентрация на ничто — активно выгонять мысли. Сложнее, но работает напрямую. Длительность входа: 1–2 минуты.

Состояние заканчивается само — как только начинаешь думать. Сознание возвращается и снова блокирует рефлексы.

---

Три режима использования

Момент — войти в состояние, и прямо сейчас всё идёт лучше. Удача, точность, скорость реакции. Пример: воздушный змей затянуло прямо в руки при сильном ветре. Персонаж в игре играл намного лучше когда управление было "отдано" Ниндзя.

Обучение — войти в состояние, сделать действие, выйти. Тело запоминает паттерн через подсознание и воспроизводит его потом уже без состояния. Это быстрее чем обычное повторение — паттерн записывается напрямую в рефлексы. Пример: несколько секунд очень хорошей игры в футбол, хотя навыка не было.

Генерация — в состоянии Ниндзя подсознание само придумывает действия, которые сознание бы не выбрало. После них — новые способности. Подсознание здесь не исполнитель, а автор. Пример: кунг-фу на пляже — тело давало странные упражнения, после которых появлялись новые возможности.

---

Эффект выхода

После выхода из состояния с вероятностью ~75% наступает временная полоса невезения. Причина неизвестна. Возможно: сознание после отключения начинает мешать сильнее обычного — перекомпенсация.

По опыту — случаев когда механика не сработала нет.

Так вот, они вместе прожили немало дней, как на планете появились злодеи. Ричард их убивал, брал в плен, и остальные ему помогали.

Ниндзи решили устроить колоссальный морской бой. Такая игра называлась «морской бой» — потому что главный ниндзя садился в маленький кораблик, а остальные нападали на него и брали на абордаж. Ни разу у них это не получалось, но они всё равно пытались.

Ричард сказал, во что они будут играть, и расставил роли:

Маг Соколин был гарпунёром — он должен был с помощью пушки зацепить палку с крюком за противоположный корабль. Его друг Бэн Соммерс бросал якоря. Вы думаете, что бросать якоря это легко? Но когда путешествуешь на большом корабле — надо резко поворачивать, бросая якоря, как это не делают на торговых судах. Чжэнь Ши был абордажником. Он стоял на абордажной доске и должен был в случае чего идти на абордаж или хотя бы стрелять из маленьких пушек.

Ну а новенький, который пришёл к ним буквально сегодня, был не настоящим ниндзей, а простым королевским шпагоносцем — и должен был быть капитаном! Потому что капитану все должны были повиноваться, а Ричард хотел проверить, сможет ли тот стать ниндзей. А тот очень хотел.

Тот развернулся от удара — и Ричард ударил внутренней стороной щита по голове Чжэнь Ши. Тот отлетел и упал на палубу. В ту же секунду Ричард отпрыгнул и закрыл абордажную доску. Корма приподнялась — и корабль перелетел через флагманский. Захватив Чжэня, всё ещё висящего на мачте, затем по вине гарпуна сбил гарпунщика под корабль, зацепил Чжэнь Ши гарпуном за мачту и приземлился рядом с флагманским кораблём.

Соколин быстро вынырнул, но его корабль уплыл. Чжэнь Ши спрыгнул на место гарпунёра, а Соколин приплыл обратно.

Чжэнь Ши отскочил, но гарпун, висящий на мачте, от удара закрутился вокруг него — и тот, на этот раз практически безвылазно, повис на мачте. Гарпун, который был довольно длинный, также замотал и Соколина после второго удара.

Помощник Бэн Соммерс в этот момент бросал якорь для остановки, но Ричард отбил якорь — и Бэн Соммерс оказался замотан в него. Тот поставил палку, но тело немного задело, и он оказался слегка замотан. Большая часть, впрочем, досталась палке.

def make_mat(name, color, roughness=0.6, metallic=0.0, emission=None):
    m = bpy.data.materials.new(name)
    m.use_nodes = True
    nodes = m.node_tree.nodes
    links = m.node_tree.links
    nodes.clear()
    out  = nodes.new('ShaderNodeOutputMaterial')

О чём это

Гравитация работает не как у Ньютона (слабее по мере удаления), а наоборот — как пружина: чем дальше объект от центра ядра, тем сильнее его притягивает. При выходе за границу гравитационного поля — притяжение мгновенно падает до нуля. Каждый отдельный объект — это своё собственное ядро.

Проблема / вопрос

Существующая ньютоновская модель описывает гравитацию как силу убывающую с расстоянием (1/r²). Но что если это неверно — и гравитация внутри поля объекта нарастает к краям, а за полем вообще исчезает? Как тогда описать притяжение между телами и можно ли управлять гравитацией?

Гипотеза

Гравитационное давление прямо пропорционально массе большего объекта и расстоянию между ними (по аналогии с законом Гука), и равно нулю за пределами максимального радиуса поля. Результат измеряется в кг и зависит от радиуса меньшего объекта.

Как устроена теория

Основные понятия
  • Ядро — группа протонов, объединивших свои силы в одно целое. Любой отдельный объект является собственным ядром.
  • Поле гравитации — зона вокруг ядра, за пределами которой притяжение мгновенно = 0.
  • k макс. — максимальный радиус гравитационного поля объекта.
  • Гравитон (e+) — частица, сам является источником поля, усиливает гравитацию.
  • Антигравитон (e-) — электрон или газ с ним, нейтрализует гравитацию.
  • Тёмная материя — скопление гравитонов.
  • Тёмная энергия — антигравитон на уровне Вселенной.
Свойства поля
  • Форма — повторяет форму объекта (не всегда шар)
  • Движение — поле движется вместе с объектом
  • Вращение — поле не вращается вместе с объектом
  • Источник — вся масса объекта целиком
  • Граница — на k_макс гравитоны рассеиваются, поле обрывается мгновенно
  • Скорость — поле распространяется мгновенно
  • Блокировка — поле проходит сквозь любое вещество, заблокировать нельзя
  • Пересечение — поля двух объектов перекрываются, силы складываются
  • Разрушение — если объект разбивается, поле делится между кусками
  • Нулевая гравитация — только у тел без массы
  • Антигравитон — при достаточном количестве полностью отменяет поле
  • Свет — отклоняется в гравитационном поле
  • Температура — не влияет на поле (влияет только масса)
  • Чёрная дыра — имеет огромное поле
  • Гравитационные волны — поток гравитонов (e+)
  • Равные массы — оба объекта создают поля одновременно, силы складываются
  • Орбиты — баланс пружинной силы и скорости планеты
Формулы

Формула 1 — сила притяжения меньшего объекта:

если k ≤ k_макс:  F = k / k_макс * m1
если k > k_макс:  F = 0
  • F — сила притяжения (кг)
  • k — расстояние между объектами (м)
  • k_макс — максимальный радиус поля меньшего объекта (м)
  • m1 — масса большего объекта, находящегося в поле меньшего (кг)
  • Меньший объект создаёт поле. Большой объект в этом поле тянет меньшего обратно.
  • k / k_макс — доля пути до края поля (0 у центра → 1 у границы): пружинное поведение
  • За пределами поля (k > k_макс): F = 0 — связь обрывается
  • Противоположность Ньютону/Эйнштейну: вместо G m1 m2 / r² (убывает) → k / k_макс * m1 (нарастает)

Природный аналог: в QCD глюонное поле тоже не расходится сферически — оно сжимается в трубку (flux tube). Аналогичная геометрия.

Формула 2 — максимальный радиус поля:

k_макс = √(m² + r²)
  • k_макс — граница поля объекта (м)
  • m — масса большего объекта (кг)
  • r — радиус объекта (м)
Квантовый уровень — "Ядерная теория"

Протон — наименьшее ядро и потенциальный создатель гравитации. Гравитация порождается протонами внутри любого тела.

Космологический уровень

Вселенная (Бог/Вселенная) — одно гигантское ядро из гуглов протонов. Внутри него разное давление гравитации из-за постоянного поглощения кварков из Поля кварков, окружающего это ядро. Именно это давление формирует разные объекты внутри Вселенной.

Антигравитон
  • Антигравитон = электрон (e-) или газ с преобладанием отрицательных ионов.
  • Условие работы: отрицательных ионов должно быть в 2 раза больше положительных.
  • Выпустить антигравитон под объектом → объект перестаёт притягиваться и поднимается вверх.
  • При выпуске антигравитона вокруг объекта образуется силовое поле, выталкивающее другие тела.
  • Чередование гравитона (e+) и антигравитона (e-) позволяет свободно перемещать объекты вверх-вниз. Устройство: палка с двумя кранами для газов.
  • Гравитационная бомба — устройство, выпускающее газ гравитон, усиливающий притяжение в радиусе вокруг себя.

Доказательства и примеры

  1. Аналогия с законом Гука: пружина растягивается сильнее по мере удаления — так же и гравитация по этой модели.
  2. Наименьшее ядро — протон: именно протоны задают гравитацию в веществе.
  3. Пружинный механизм реально существует в природе на всех масштабах: конфайнмент кварков (QCD), гармонический осциллятор в атоме, кристаллические решётки, самогравитация планет — везде один и тот же паттерн (см. раздел "Универсальный принцип равновесия").

Что из этого следует

  • Гравитация управляема: можно создать устройство с газом антигравитона для левитации объектов.
  • У каждого объекта есть конечная зона притяжения — за ней он не влияет на окружающее.
  • Вселенная — не бесконечно равномерная, а живая структура с переменным внутренним давлением гравитации.

Связи с другими теориями

  • Закон Гука (F = k·x) — прямой аналог: сила пружины растёт с расстоянием. Здесь то же самое для гравитации.
  • Закон всемирного тяготения Ньютона (F ~ 1/r²) — противоположная зависимость от расстояния. Теория Луки её опровергает внутри поля.
  • Общая теория относительности Эйнштейна — описывает гравитацию через искривление пространства-времени, не через пружину.
  • Стандартная модель — протон действительно состоит из кварков, что перекликается с "ядерной теорией".

Пружинное поведение (сила растёт с расстоянием + обнуление за границей) — универсальный принцип природы. Один и тот же паттерн встречается на каждом масштабе:

1. Орбиты планет противоречат пружинной модели

Если гравитация растёт с расстоянием — планеты не могут стабильно орбитировать. Чем дальше планета от Солнца, тем сильнее её тянет — она должна ускоряться и улетать, а не двигаться по орбите. На практике всё наоборот: Юпитер движется медленнее Земли (13 км/с против 30 км/с). Это прямо противоречит пружинному росту.

Нужно объяснить: как стабильные орбиты возможны при нарастающей с расстоянием гравитации?

2. Единицы измерения ✅ РЕШЕНО

Формула P = b k / (2 r) даёт: кг × м / м = кг. Настоящая единица, не изобретённая. Единица Л (кг·м) была в старой формуле без r — больше не актуальна.

---

5. Двойка в знаменателе — нет обоснования

Почему именно /2? В законе Гука нет двойки. Это не следует из аналогии с пружиной. Нужно либо объяснить физический смысл, либо это просто коэффициент подгонки.

---

6. Мгновенное обнуление поля — нет механизма 🟡 ЧАСТИЧНО

За пределами k_макс гравитация мгновенно = 0. Резкие границы полей существуют в природе — QCD (конфайнмент кварков), электрическое поле внутри сферы, ядерный потенциал. Так что резкий обрыв физически допустим. Но механизм именно для гравитации всё ещё не объяснён.

---

7. Антигравитон = электрон — не работает так

Электрон несёт отрицательный электрический заряд, а не антигравитацию. Выпустить газ с отрицательными ионами под объектом — это создать электрическое поле, а не гравитационное. Электромагнетизм и гравитация — разные взаимодействия. Нужно объяснить механизм связи между зарядом и гравитацией в этой теории.

---

8. Формула 1 не зависит от массы меньшего объекта

P = b k / (2 r) — масса меньшего объекта не входит, только его радиус r. Давление Земли на Луну и на булыжник того же радиуса на том же расстоянии — одинаковое. Масса притягиваемого объекта не имеет значения?

---

9. Формула 2 даёт нереально огромные поля ✅ РЕШЕНО

Старая формула b^g/g давала поля в 10²³ раз больше Вселенной. Исправлена на b^(1/g):

  • Протон: 4.1×10⁻¹⁴ м (атомный масштаб)
  • Земля: 2.4×10¹² м (2.4 млрд км — в пределах Солнечной системы)
  • Солнце: 1.4×10¹⁵ м (1.4 трлн км)

Все значения находятся внутри наблюдаемой Вселенной.

---

13. Единицы формулы 2 ✅ РЕШЕНО

k_макс = b^(1/g) — b в кг, результат должен быть в метрах. Формально кг^0.5 ≠ метры.

Решение: формула работает в СИ через неявное деление на опорные единицы:

k_макс [м] = (b [кг] / 1 [кг])^(1/g) × 1 [м]

Делим на 1 кг → безразмерное число → возводим в степень → умножаем на 1 м. Единицы сходятся. Это стандартный приём для эмпирических формул — работает только в СИ.

---

14. g = 2.0006 не универсальна 🔴 ОТКРЫТО

Проверено: разные тела требуют разных g для совпадения формулы 2 с реальными границами поля.

Нужно: либо объяснить почему g меняется от тела к телу, либо найти формулу g(b), либо признать что формула 2 работает только для звёздных масс.

10. Нет объяснения почему мы не чувствуем гравитацию далёких звёзд

По модели, поле каждой звезды огромно — значит Земля находится в полях миллиардов звёзд одновременно, и все они тянут в разные стороны с нарастающей с расстоянием силой. Тогда суммарное воздействие должно быть колоссальным и ненаправленным — мы бы это чувствовали.

Три пункта в разделе "Доказательства":

  1. Аналогия с законом Гука — аналогия это не доказательство

12. P не совпадает с реальным весом ✅ РЕШЕНО

Старая формула P = bk/(2r) давала результат в 10²⁹ раз больше реального веса.

Решение: Формула 1 переосмыслена. P — не вес, а масса меньшего объекта, вычисленная из поля: P = b * k² (в СИ, неявное / 1 м²). Масса — это ответ формулы, входных данных о меньшем объекте не нужно.
